  process.env.TWILIO_ACCOUNT_SID,
  process.env.TWILIO_AUTH_TOKEN
);

/**
 * POST /webhook/whatsapp
 * Receive incoming WhatsApp messages from Twilio
 */
exports.receiveMessage = async (req, res) => {
  try {
    const startTime = Date.now();
    
    // Extract message data from Twilio webhook
    const {
      Body: messageBody,
      From: senderNumber,
      To: recipientNumber,
      MessageSid: messageId,
      NumMedia: mediaCount
    } = req.body;

    logger.info(`📩 Incoming message from ${senderNumber}: ${messageBody?.substring(0, 50)}...`);

    // Validate required fields
    if (!messageBody || !senderNumber) {
      return res.status(400).send('Missing required fields');
    }

    // Detect language of incoming message
    const detectedLang = detectLanguage(messageBody);
    logger.info(`🌍 Detected language: ${detectedLang}`);

    // Get or create user conversation history
    const conversationHistory = await getUserConversation(senderNumber);
    
    // Save incoming message to database
    await saveMessage({
      messageId,
      senderNumber,
      recipientNumber,
      messageBody,
      direction: 'inbound',
      language: detectedLang,
      mediaCount: parseInt(mediaCount) || 0
    });

    // Emit real-time event to admin dashboard
    const io = req.app.get('io');
    io.to('admin-dashboard').emit('new-message', {
      from: senderNumber,
      message: messageBody.substring(0, 100),
      timestamp: new Date(),
      language: detectedLang
    });

    // Send typing indicator to user
    await sendTypingIndicator(senderNumber);

    // Process with AI
    const aiResponse = await getAIResponse({
      message: messageBody,
      userId: senderNumber,
      language: detectedLang,
      history: conversationHistory
    });

    // Translate response if needed (respond in user's language)
    let finalResponse = aiResponse;
    if (detectedLang !== 'en') {
      finalResponse = await translate(aiResponse, detectedLang);
    }

    // Send reply via Twilio WhatsApp
    await sendMessage(senderNumber, finalResponse);

    // Save outgoing message
    await saveMessage({
      messageId: `out_${Date.now()}`,
      senderNumber: recipientNumber,
      recipientNumber: senderNumber,
      messageBody: finalResponse,
      direction: 'outbound',
      language: detectedLang,
      processingTime: Date.now() - startTime
    });

    // Log metrics
    const processingTime = Date.now() - startTime;
    logger.info(`✅ Response sent in ${processingTime}ms`);

    // Respond to Twilio webhook immediately (async processing)
    res.status(200).send(`
      
      
    `);

  } catch (error) {
    logger.error('❌ Webhook error:', error);
    
    // Send error fallback message
    try {
      await sendMessage(req.body.From, 'Maaf, sedang ada gangguan. Silakan coba lagi nanti.');
    } catch (sendError) {
      logger.error('Failed to send error message:', sendError);
    }
    
    res.status(200).send('');
  }
};

/**
 * Get AI response using GPT-4
 */
async function getAIResponse({ message, userId, language, history = [], persona = 'default' }) {
  try {
    // Build conversation messages
    const messages = [
      { 
        role: 'system', 
        content: `${SYSTEM_PROMPTS[persona]}\n\nBahasa utama: ${getLanguageName(language)}` 
      },
      
      // Add conversation history for context (last 10 messages)
      ...history.slice(-10).map(msg => ({
        role: msg.direction === 'inbound' ? 'user' : 'assistant',
        content: msg.messageBody
      })),
      
      // Current user message
      { role: 'user', content: message }
    ];

    // Call OpenAI API
    const completion = await openai.chat.completions.create({
      model: process.env.OPENAI_MODEL || 'gpt-4',
      messages: messages,
      max_tokens: parseInt(process.env.MAX_TOKENS) || 500,
      temperature: parseFloat(process.env.AI_TEMPERATURE) || 0.7,
      top_p: 1,
      frequency_penalty: 0,
      presence_penalty: 0,

      // Function calling capabilities (for actions like orders, payments)
      functions: [
        {
          name: 'create_order',
          description: 'Create a new product order',
          parameters: {
            type: 'object',
            properties: {
              product_id: { type: 'string', description: 'Product ID' },
              quantity: { type: 'integer', description: 'Quantity to order' },
              customer_name: { type: 'string' },
              address: { type: 'string' }
            },
            required: ['product_id', 'quantity']
          }
        },
        {
          name: 'check_product_availability',
          description: 'Check if product is in stock',
          parameters: {
            type: 'object',
            properties: {
              product_name: { type: 'string' }
            },
            required: ['product_name']
          }
        },
        {
          name: 'initiate_payment',
          description: 'Start payment process for an order',
          parameters: {
            type: 'object',
            properties: {
              order_id: { type: 'string' },
              payment_method: { 
                type: 'string', 
                enum: ['gopay', 'ovo', 'dana', 'bank_transfer', 'qris'] 
              }
            },
            required: ['order_id', 'payment_method']
          }
        }
      ],
      function_call: 'auto'
    });

    const response = completion.choices[0].message;

    // Check if AI wants to call a function
    if (response.function_call) {
      return await handleFunctionCall(response.function_call, userId);
    }

    // Log token usage for cost tracking
    logger.info(`🤖 OpenAI API call - Tokens used: ${completion.usage.total_tokens}, Cost: $${(completion.usage.total_tokens * 0.00003).toFixed(4)}`);

    return response.content;

  } catch (error) {
    logger.error('OpenAI API Error:', error);
    
    // Fallback responses based on error type
    if (error.code === 'rate_limit_exceeded') {
      return 'Maaf, sedang banyak yang bertanya. Silakan tunggu sebentar lalu coba lagi ya! 😊';
    }
    
    if (error.code === 'insufficient_quota') {
      return 'Maaf, layanan AI sedang maintenance. Tim kami akan segera menghubungi Anda.';
    }
    
    return 'Maaf, terjadi kesalahan. Silakan coba lagi atau hubungi admin.';
  }
}

/**
 * Handle AI function calls (orders, payments, etc.)
 */
async function handleFunctionCall(functionCall, userId) {
  const { name, arguments: args } = functionCall;
  const parsedArgs = JSON.parse(args);

  logger.info(`🔧 Function called: ${name}`, parsedArgs);

  switch (name) {
    case 'create_order':
      // Import and call order service
      const { createOrder } = require('./orderService');
      return await createOrder({ ...parsedArgs, userId });
    
    case 'check_product_availability':
      const { checkStock } = require('./productService');
      return await checkStock(parsedArgs.product_name);
    
    case 'initiate_payment':
      const { createPayment } = require('./paymentService');
      return await createPayment(parsedArgs);
    
    default:
      return 'Maaf, fitur tersebut sedang dalam pengembangan.';
  }
}

const jwt = require('jsonwebtoken');
const { promisify } = require('util');

const JWT_SECRET = process.env.JWT_SECRET || 'your-super-secret-key-change-in-production';
const JWT_EXPIRES_IN = process.env.JWT_EXPIRES_IN || '7d';
const JWT_COOKIE_EXPIRES = 7; // days

/**
 * Generate JWT token
 */
const signToken = (id, role = 'user') => {
  return jwt.sign(
    { id, role },
    JWT_SECRET,
    { expiresIn: JWT_EXPIRES_IN }
  );
};

/**
 * Create and send token in cookie
 */
const createSendToken = (user, statusCode, res) => {
  const token = signToken(user.id, user.role);

  const cookieOptions = {
    expires: new Date(Date.now() + JWT_COOKIE_EXPIRES * 24 * 60 * 60 * 1000),
    httpOnly: true,
    secure: process.env.NODE_ENV === 'production',
    sameSite: 'strict'
  };

  res.cookie('jwt', token, cookieOptions);

  // Remove password from output
  user.password = undefined;

  res.status(statusCode).json({
    status: 'success',
    token,
    data: { user }
  });
};

/**
 * Protect routes - Verify JWT token
 */
const authenticateJWT = async (req, res, next) => {
  try {
    // 1) Get token from header or cookie
    let token;
    if (req.headers.authorization && req.headers.authorization.startsWith('Bearer')) {
      token = req.headers.authorization.split(' ')[1];
    } else if (req.cookies.jwt) {
      token = req.cookies.jwt;
    }

    if (!token) {
      return res.status(401).json({
        status: 'error',
        message: 'Anda belum login. Silakan login untuk mengakses.'
      });
    }

    // 2) Verify token
    const decoded = await promisify(jwt.verify)(token, JWT_SECRET);

    // 3) Check if user still exists (optional - requires DB)
    // const currentUser = await User.findById(decoded.id);
    // if (!currentUser) {
    //   return res.status(401).json({
    //     status: 'error',
    //     message: 'User tidak lagi exist.'
    //   });
    // }

    // Grant access
    req.user = decoded;
    next();

  } catch (error) {
    return res.status(401).json({
      status: 'error',
      message: 'Token invalid atau expired.'
    });
  }
};

/**
 * Restrict to specific roles
 */
const restrictTo = (...roles) => {
  return (req, res, next) => {
    if (!roles.includes(req.user.role)) {
      return res.status(403).json({
        status: 'error',
        message: 'Anda tidak memiliki izin untuk aksi ini.'
      });
    }
    next();
  };
};

/**
 * Optional auth - attach user if token exists, but don't require it
 */
const optionalAuth = async (req, res, next) => {
  try {
    let token;
    if (req.headers.authorization?.startsWith('Bearer')) {
      token = req.headers.authorization.split(' ')[1];
    }
    
    if (token) {
      const decoded = await promisify(jwt.verify)(token, JWT_SECRET);
      req.user = decoded;
    }
    next();
  } catch (error) {
    next();
  }
};

const { PrismaClient } = require('@prisma/client');
const logger = require('../utils/logger');

// Create Prisma instance with logging in development
const prisma = new PrismaClient({
  log: process.env.NODE_ENV === 'development' 
    ? ['query', 'info', 'warn', 'error'] 
    : ['error']

/**
 * Connect to database
 */
async function connectDB() {
  try {
    await prisma.$connect();
    logger.info('✅ Database connected successfully');
    
    // Test query
    await prisma.$queryRaw`SELECT 1`;
    
    return prisma;
  } catch (error) {
    logger.error('❌ Database connection failed:', error.message);
    throw error;
  }
}

/**
 * Graceful disconnect
 */
async function disconnectDB() {
  await prisma.$disconnect();
  logger.info('Database disconnected');
}

/**
 * Health check
 */
async function healthCheck() {
  try {
    await prisma.$queryRaw`SELECT 1`;
    return { status: 'healthy', database: 'connected' };
  } catch (error) {
    return { status: 'unhealthy', error: error.message };
  }
}

// Translation cache (Redis recommended for production)
const translationCache = new Map();

/**
 * Detect language from text
 */
function detectLanguage(text) {
  if (!text || typeof text !== 'string') return 'id'; // Default to Indonesian

  const cleanText = text.toLowerCase().trim();

  // Check each language pattern
  let bestMatch = { lang: 'id', score: 0 };

  for (const [lang, config] of Object.entries(LANGUAGE_PATTERNS)) {
    let score = 0;

    // Check starting patterns
    for (const pattern of config.patterns) {
      if (pattern.test(cleanText)) score += 0.4;
    }

    // Count common words
    const words = cleanText.split(/\s+/);
    const commonWordMatches = words.filter(w => config.commonWords.includes(w)).length;
    score += (commonWordMatches / Math.max(words.length, 1)) * 0.6;

    if (score > bestMatch.score) {
      bestMatch = { lang, score };
    }
  }

  return bestMatch.lang;
}

/**
 * Translate text using OpenAI (more accurate than Google Translate for context)
 */
async function translate(text, targetLang, sourceLang = null) {
  const cacheKey = `${sourceLang || 'auto'}:${targetLang}:${text.substring(0, 100)}`;
  
  // Check cache first
  if (translationCache.has(cacheKey)) {
    return translationCache.get(cacheKey);
  }

  try {
    // Use OpenAI for contextual translation
    const OpenAI = require('openai');
    const openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY });

    const targetNames = {
      id: 'Bahasa Indonesia',
      en: 'English',
      jv: 'Basa Jawa (Javanese)',
      su: 'Basa Sunda (Sundanese)',
      ar: 'العربية (Arabic)',
      zh: '中文 (Mandarin Chinese)',
      ja: '日本語 (Japanese)',
      ko: '한국어 (Korean)',
      es: 'Español (Spanish)',
      fr: 'Français (French)'
    };

    const prompt = sourceLang 
      ? `Translate the following text from ${sourceLang.toUpperCase()} to ${targetNames[targetLang]}. Only output the translated text, nothing else:\n\n"${text}"`
      : `Detect the language of the following text and translate it to ${targetNames[targetLang]}. Only output the translated text:\n\n"${text}"`;

    const response = await openai.chat.completions.create({
      model: 'gpt-3.5-turbo', // Cheaper model for translations
      messages: [{ role: 'user', content: prompt }],
      max_tokens: Math.min(text.length * 2, 1000),
      temperature: 0.3 // Lower temperature for more consistent translations
    });

    const translated = response.choices[0].message.content.trim();
    
    // Cache result (with TTL)
    translationCache.set(cacheKey, translated);
    setTimeout(() => translationCache.delete(cacheKey), 3600000); // 1 hour TTL

    return translated;

  } catch (error) {
    console.error('Translation error:', error);
    return text; // Return original on error
  }
}
